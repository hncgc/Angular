function setCookie(name,key){

var date=new Date(); 
date.setTime(date.getTime()+60*60*1000); //设置date为当前时间+60分
$.cookie(name, key, {path: "/", expires: date});
}

//传入要获取的键

function getCookie(name){
return $.cookie(name);
}
